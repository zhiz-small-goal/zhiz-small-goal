cmake_minimum_required(VERSION 3.20)
project(commented_stats VERSION 1.0 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

add_library(stats STATIC 
    stats.cpp 
    stats.h
)

add_executable(commented_stats 
    main.cpp
    stats.cpp
    stats.h
)

target_link_libraries(commented_stats PRIVATE stats)
